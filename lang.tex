\section{Language definition}\label{sec:lang}

\begin{center}
 \begin{small}
  \begin{grammar}
   \production{\re}{\none \mid \eps \mid \atom \mid \re_0\catop\re_1 \mid \re_0\orop\re_1 \mid \starop{\re}\mid \re_0\shuffleop\re_1 \mid \var \mid \existOp{\var}{\cts}{\re}}{} \\
   \production{\cts}{\ct_1,\ldots,\ct_n}{\hspace*{-5em}with $\ct_1,\ldots,\ct_n\in\cset$} \\
   \production{\atom}{\predSym(\eventTerm_1,...\eventTerm_n)}{\hspace*{-5em}with $\predSym\in\predSet$} \\
   \production{\eventTerm}{\val \mid \var}{\hspace*{-5em}with $\val\in\vset, \var\in\varSet$} \\
  \end{grammar}
 \end{small}
\end{center}
